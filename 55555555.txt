while 1:
    inp=raw_input()
    wordList=[]
    NotWordList=[]
 
    for i in inp:
        if(64<ord(i)<91 or 96<ord(i)<123):
            NotWordList.append('A')
            wordList.append(i)
        else:
            NotWordList.append(i)
 
    wordList=sorted(wordList,key=lambda x:x.lower())
    count=0
    #out_word=[]
    for i in xrange(len(NotWordList)):
        if NotWordList[i]=='A':
            NotWordList[i]=wordList[count]
            count+=1
    print ''.join(NotWordList)
except:
        break
